1. List all customers along with their sales representative's first name and last name (equi join on `salesRepEmployeeNumber`).
    SELECT 
    c.customerName, 
    e.firstName, 
    e.lastName
    FROM customers c, employees e
    WHERE c.salesRepEmployeeNumber = e.employeeNumber
 

2. Find all orders placed by customers from 'USA'. Show `orderNumber`, `customerName`, and `orderDate`.
   select c.customerName, o.ordernumber, o.orderdate from customers c, orders o where c.customerNumber = o.customerNumber and c.country = 'usa'

3. Display product details (`productName`, `productLine`) along with the product line's text description.
   select p.productName,p.productLine,pd.textdescription from products p,productLines pd where p.productline = pd.productline

4. Show all employees and their office city and country.

9. List employees who report to 'Diane Murphy' (use equi join on `reportsTo`).
   SELECT 
    e.employeeNumber AS 'Employee ID',
    e.firstName AS 'First Name',
    e.lastName AS 'Last Name',
    e.jobTitle AS 'Job Title',
    m.firstName AS 'Manager First Name',
    m.lastName AS 'Manager Last Name'
FROM employees e, employees m 
WHERE e.reportsTo = m.employeeNumber -- આ Equi-Join શરત છે
  AND m.firstName = 'Diane' 
  AND m.lastName = 'Murphy';

21. Find the average credit limit of customers by country.
    SELECT 
    c.country,c.customerName,c.customerNumber, 
    AVG(c.creditLimit) AS average_credit_limit 
    FROM 
    customers c
    INNER JOIN 
    orders o ON c.customerNumber = o.customerNumber
    GROUP BY 
    c.country 
    LIMIT 0, 1000;

24. Count how many products are in each product line.
    
SELECT 
    pl.productLine,
    COUNT(p.productCode) AS total_products 
FROM 
    productlines pl
LEFT JOIN 
    products p ON pl.productLine = p.productLine
GROUP BY 
    pl.productLine;

26. Find customers who have placed more than 5 orders.
    select c.customerNumber,c.customerName,count(o.ordernumber) as total_orders from customers c inner join orders o on c.customerNumber = o. customerNumber group by c.customerNumber,c.customerName HAVING count(o.orderNumber) > 5

27. Show product lines where the average MSRP is greater than 100.
SELECT 
    pl.productLine, 
    AVG(p.MSRP) AS average_msrp 
FROM 
    productlines pl
INNER JOIN 
    products p ON pl.productLine = p.productLine 
GROUP BY 
    pl.productLine 
HAVING 
    AVG(p.MSRP) > 100;

41. For each sales rep, show the total sales amount they generated (join `employees` → `customers` → `orders` → `orderdetails`). Sort by total sales descending.
select e.employeeNumber,e.firstName,e.lastname,sum(od.priceEach * quantityOrdered) as total_sales_amount from employees e inner join customers c on e.employeeNumber = c.salesRepEmployeeNumber inner join orders o on o.customerNumber = c.customerNumber inner join orderdetails od on o.ordernumber = od.ordernumber group by e.employeeNumber,e.firstName,e.lastName order by total_sales_amount desc
