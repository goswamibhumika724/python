1. List all customers along with their sales representative's first name and last name (equi join on `salesRepEmployeeNumber`).
    SELECT 
    c.customerName, 
    e.firstName, 
    e.lastName
    FROM customers c, employees e
    WHERE c.salesRepEmployeeNumber = e.employeeNumber
 

2. Find all orders placed by customers from 'USA'. Show `orderNumber`, `customerName`, and `orderDate`.
   select c.customerName, o.ordernumber, o.orderdate from customers c, orders o where c.customerNumber = o.customerNumber and c.country = 'usa'

3. Display product details (`productName`, `productLine`) along with the product line's text description.
   select p.productName,p.productLine,pd.textdescription from products p,productLines pd where p.productline = pd.productline

4. Show all employees and their office city and country.
   select e.firstname, e.lastname, o.officecode, o.city, o.country  from employees e, offices o where e.officecode = o.officecode

5. List all payments with the corresponding customer name and amount.
   select c.customerName,p.amount from customers c, payments p where c.customerNumber = p.customerNumber

6. Find all order details (`orderNumber`, `productCode`, `quantityOrdered`) with the product name.
   select o.ordernumber,o.productcode,o.quantityOrdered,p.productName from orderdetails o, products p where o.productCode = p.productCode

7. Show customers from 'France' and their sales rep's job title.
   select c.customerName,c.country,e.firstName,e.lastName,e.jobtitle from customers c, employees e where c.salesRepEmployeeNumber = e.employeeNumber
  AND c.country = 'France'

8. Display all products in the 'Classic Cars' product line with their buy price and MSRP.
   select p.productName ,p.productLine,p.buyprice,p.MSRP from productLines pd, products p where p.productline = pd.productline and p.productLine = 'Classic Cars' 

9. List employees who report to 'Diane Murphy' (use equi join on `reportsTo`).
   SELECT 
    e.employeeNumber AS 'Employee ID',
    e.firstName AS 'First Name',
    e.lastName AS 'Last Name',
    e.jobTitle AS 'Job Title',
    m.firstName AS 'Manager First Name',
    m.lastName AS 'Manager Last Name'
FROM employees e, employees m 
WHERE e.reportsTo = m.employeeNumber -- આ Equi-Join શરત છે
  AND m.firstName = 'Diane' 
  AND m.lastName = 'Murphy';

10. Show all orders that are 'Shipped' along with the customer name.
    select c.customerName, o.orderNumber,o.shippeddate,o.status from customers c, orders o where c.customerNumber = o.customerNumber and o.status = 'Shipped'


11. List all customers and their sales representative's name (use LEFT JOIN). Include customers who have no sales rep.
    select c. customerNumber, c.customerName,c.salesRepEmployeeNumber,e.firstName as repfirstname,e.lastname as replastname from customers c left join employees e on 
c.salesRepEmployeeNumber = e.employeeNumber  

12. Show all employees and the customers they manage (use LEFT JOIN). Include employees with no customers.
    select e.employeeNumber,e.firstName,e.lastName,c.customerNumber,c.customerName from employees e left join customers c on e.employeeNumber = c.salesRepEmployeeNumber

13. List all products and how many times they have been ordered (use LEFT JOIN with `orderdetails`).
   select p.productcode,p.productName,count(od.orderNumber) as timesordered from products p left join orderdetails od on p.productcode = od.productCode group by p.productname,p.productcode

14. Show all offices and the number of employees in each office (use LEFT JOIN).
    select o.officecode,o.city,o.country,e.firstname,e.lastname,e.employeeNumber from offices o left join employees e on o.officeCode = e.officeCode

15. Display all customers and their total payment amount (use LEFT JOIN with `payments`).
    select c.customerNumber,c.customerName,p.amount as total_payment_amount from customers c left join payments p on c.customerNumber = p.customerNumber

16. Use RIGHT JOIN to show all employees and any customers they represent.
    select  c.salesRepEmployeeNumber,e.employeeNumber,e.firstName,e.lastName from customers c right join employees e on c.salesRepEmployeeNumber = e.employeeNumber

17. Use LEFT JOIN to list all product lines and the products in them.
    select p.productline,pd.productCode,pd.productname from productlines p left join products pd on p.productline = pd.productline

18. Show all orders and the customer name, including orders without a matching customer (if any).
    select o.orderNumber,c.customerName,c.customerNumber from orders o left join customers c on o.customerNumber = c.customerNumber
 
19. Find the total number of orders per customer. Show `customerName` and order count.
    select c.customerNumber,c.customerName,count(o.ordernumber) as total_number_orders from customers c left join orders o on c.customerNumber = o.customerNumber group by c.customerName,c.customerNumber

20. Calculate total sales amount (sum of `priceEach * quantityOrdered`) per product line.
    select p.productcode,p.productline,p.productname,(o.priceEach * quantityOrdered) from orderdetails o right join products p on o.productcode = p.productcode
